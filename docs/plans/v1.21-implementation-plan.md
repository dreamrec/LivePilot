def _infer_move_family(target) -> str:
    """Determine the move_class for the ledger entry.

    Priority: seed.family → inspect compiled_plan[0].tool → default mix.
    """
    # Priority 1: explicit seed family
    seed = getattr(target, "seed", None)
    if seed is not None and getattr(seed, "family", None):
        return seed.family

    # Priority 2: inspect first step's tool and map
    plan = getattr(target, "compiled_plan", None) or {}
    steps = plan.get("steps", [])
    if steps:
        first_tool = steps[0].get("tool", "")
        return _TOOL_TO_FAMILY.get(first_tool, "mix")

    return "mix"  # safe default

def _compile_configure_record_readiness(move: SemanticMove, kernel: dict) -> CompiledPlan:
    args = kernel.get("seed_args") or {}
    track_index = args.get("track_index")
    armed = args.get("armed")
    exclusive = args.get("exclusive", False)

    if track_index is None or armed is None:
        return _empty_plan(move, [
            "configure_record_readiness requires seed_args.track_index + armed"
        ])
    if not isinstance(track_index, int):
        return _empty_plan(move, [f"track_index must be int, got {track_index!r}"])
    if not isinstance(armed, bool):
        return _empty_plan(move, [f"armed must be bool, got {armed!r}"])
    if not isinstance(exclusive, bool):
        return _empty_plan(move, [f"exclusive must be bool, got {exclusive!r}"])
    if exclusive and not armed:
        return _empty_plan(move, [
            "exclusive=True requires armed=True (the point of exclusive "
            "is to become the single armed track); to disarm individually "
            "call configure_record_readiness with exclusive=False"
        ])

    # Path: exclusive arming uses the dedicated set_exclusive_arm tool
    # which disarms all other regular tracks + arms the target in one call.
    if exclusive and armed:
        step = CompiledStep(
            tool="set_exclusive_arm",
            params={"track_index": track_index},
            description=f"Exclusive-arm track {track_index} (disarms others)",
            verify_after=True,
            backend="remote_command",
        )
    else:
        step = CompiledStep(
            tool="set_track_arm",
            params={"track_index": track_index, "armed": armed},
            description=(
                f"{'Arm' if armed else 'Disarm'} track {track_index}"
            ),
            verify_after=True,
            backend="remote_command",
        )

    return CompiledPlan(
        move_id=move.move_id,
        intent=move.intent,
        steps=[step],
        risk_level=move.risk_level,
        summary=step.description,
        requires_approval=(kernel.get("mode", "improve") != "explore"),
        warnings=[],
    )

"""Schema validator for affordance-device preset YAML files.

Enforces at test-time that every seed preset file is well-formed.
The loader (presets.py) is tolerant of missing optional fields at
runtime so the product never crashes on a bad preset; the schema
validator is the pre-ship gate.
"""

from __future__ import annotations

from pathlib import Path
from typing import Any

import yaml


_REQUIRED_TOPLEVEL = {"device_slug", "device_class_name", "presets"}
_REQUIRED_PER_PRESET = {"description", "param_overrides"}
_OPTIONAL_PER_PRESET = {"risk_notes", "suggested_pairings"}


def validate_preset_file(path: Path) -> list[str]:
    """Return a list of validation errors. Empty list = valid.

    Errors are human-readable one-liners suitable for test failure output.
    """
    errors: list[str] = []

    try:
        data = yaml.safe_load(path.read_text(encoding="utf-8"))
    except yaml.YAMLError as exc:
        return [f"{path.name}: YAML parse error — {exc}"]

    if not isinstance(data, dict):
        return [f"{path.name}: top-level must be a mapping, got {type(data).__name__}"]

    missing = _REQUIRED_TOPLEVEL - set(data.keys())
    if missing:
        errors.append(f"{path.name}: missing required top-level fields {sorted(missing)}")

    slug = data.get("device_slug")
    if slug is not None and not isinstance(slug, str):
        errors.append(f"{path.name}: device_slug must be a string")
    elif slug and slug != path.stem:
        errors.append(
            f"{path.name}: device_slug {slug!r} must match filename stem {path.stem!r}"
        )

    class_name = data.get("device_class_name")
    if class_name is not None and not isinstance(class_name, str):
        errors.append(f"{path.name}: device_class_name must be a string")

    presets = data.get("presets")
    if presets is not None:
        if not isinstance(presets, dict):
            errors.append(f"{path.name}: presets must be a mapping")
        elif not presets:
            errors.append(f"{path.name}: presets dict is empty")
        else:
            for preset_name, preset in presets.items():
                errors.extend(_validate_preset(path.name, preset_name, preset))

    return errors


def _validate_preset(filename: str, name: str, preset: Any) -> list[str]:
    errors: list[str] = []
    if not isinstance(preset, dict):
        return [f"{filename}: preset {name!r} must be a mapping"]

    missing = _REQUIRED_PER_PRESET - set(preset.keys())
    if missing:
        errors.append(
            f"{filename}: preset {name!r} missing required fields {sorted(missing)}"
        )

    description = preset.get("description")
    if description is not None and not isinstance(description, str):
        errors.append(f"{filename}: preset {name!r} description must be a string")

    overrides = preset.get("param_overrides")
    if overrides is not None:
        if not isinstance(overrides, dict):
            errors.append(f"{filename}: preset {name!r} param_overrides must be a dict")
        elif not overrides:
            errors.append(f"{filename}: preset {name!r} param_overrides is empty")
        else:
            for k, v in overrides.items():
                if not isinstance(k, str):
                    errors.append(
                        f"{filename}: preset {name!r} param_overrides key {k!r} must be a string"
                    )
                if not isinstance(v, (int, float, bool)):
                    errors.append(
                        f"{filename}: preset {name!r} param_overrides value for {k!r} "
                        f"must be number|bool, got {type(v).__name__}"
                    )

    pairings = preset.get("suggested_pairings")
    if pairings is not None:
        if not isinstance(pairings, list):
            errors.append(f"{filename}: preset {name!r} suggested_pairings must be a list")
        elif not all(isinstance(p, str) for p in pairings):
            errors.append(
                f"{filename}: preset {name!r} suggested_pairings entries must be strings"
            )

    risk_notes = preset.get("risk_notes")
    if risk_notes is not None and not isinstance(risk_notes, str):
        errors.append(f"{filename}: preset {name!r} risk_notes must be a string")

    extras = set(preset.keys()) - _REQUIRED_PER_PRESET - _OPTIONAL_PER_PRESET
    if extras:
        errors.append(
            f"{filename}: preset {name!r} has unknown fields {sorted(extras)} "
            f"(allowed: {sorted(_REQUIRED_PER_PRESET | _OPTIONAL_PER_PRESET)})"
        )

    return errors
```

"""Preset loader for affordance-device YAML files.

Runtime resolution only — schema validation lives in _schema.py and
fires at test-time.
"""

from __future__ import annotations

from pathlib import Path
from typing import Any, Optional

import yaml


_AFFORDANCES_ROOT = Path(__file__).parent / "devices"


def _load_device_yaml(device_slug: str) -> Optional[dict]:
    path = _AFFORDANCES_ROOT / f"{device_slug}.yaml"
    if not path.exists():
        return None
    try:
        data = yaml.safe_load(path.read_text(encoding="utf-8"))
    except yaml.YAMLError:
        return None
    return data if isinstance(data, dict) else None


def resolve_preset(device_slug: str, preset_name: str) -> Optional[dict[str, Any]]:
    """Return the param_overrides dict for a named preset, or None.

    Returns None on: missing device file, missing preset, unparseable YAML,
    or preset record without param_overrides.
    """
    data = _load_device_yaml(device_slug)
    if data is None:
        return None
    preset = data.get("presets", {}).get(preset_name)
    if not isinstance(preset, dict):
        return None
    overrides = preset.get("param_overrides")
    return dict(overrides) if isinstance(overrides, dict) else None


def get_preset_metadata(device_slug: str, preset_name: str) -> Optional[dict]:
    """Return the full preset record (description + pairings + risk_notes)
    or None."""
    data = _load_device_yaml(device_slug)
    if data is None:
        return None
    preset = data.get("presets", {}).get(preset_name)
    return dict(preset) if isinstance(preset, dict) else None


def list_devices() -> list[str]:
    """Return device slugs with available preset files."""
    if not _AFFORDANCES_ROOT.exists():
        return []
    return sorted(p.stem for p in _AFFORDANCES_ROOT.glob("*.yaml"))


def list_presets(device_slug: str) -> list[str]:
    """Return preset names for a given device slug."""
    data = _load_device_yaml(device_slug)
    if data is None:
        return []
    return sorted(data.get("presets", {}).keys())
```

def _compile_configure_device(move: SemanticMove, kernel: dict) -> CompiledPlan:
    args = kernel.get("seed_args") or {}
    track_index = args.get("track_index")
    device_index = args.get("device_index")
    preset_name = args.get("preset")
    device_slug = args.get("device_slug")
    explicit_overrides = args.get("param_overrides", {})

    if track_index is None or device_index is None:
        return _empty_plan(move, [
            "configure_device requires seed_args.track_index + device_index"
        ])
    if not isinstance(track_index, int) or not isinstance(device_index, int):
        return _empty_plan(move, [
            f"track_index and device_index must be ints, got "
            f"{type(track_index).__name__}/{type(device_index).__name__}"
        ])
    if device_index < 0:
        return _empty_plan(move, [f"device_index must be non-negative, got {device_index}"])
    if not isinstance(explicit_overrides, dict):
        return _empty_plan(move, [
            f"param_overrides must be a dict, got {type(explicit_overrides).__name__}"
        ])

    # Resolve preset if requested
    preset_overrides: dict = {}
    if preset_name is not None:
        if not device_slug:
            return _empty_plan(move, [
                "preset seed_arg requires device_slug (v1.21 contract). "
                "Auto-inference from class_name is v1.22 scope."
            ])
        from ...affordances import resolve_preset
        resolved = resolve_preset(device_slug, preset_name)
        if resolved is None:
            return _empty_plan(move, [
                f"No preset {preset_name!r} for device slug {device_slug!r}. "
                f"Check mcp_server/affordances/devices/{device_slug}.yaml exists "
                f"and contains the named preset."
            ])
        preset_overrides = resolved

    # Merge: preset first, explicit last-write-wins
    merged: dict = {}
    merged.update(preset_overrides)
    merged.update(explicit_overrides)

    if not merged:
        return _empty_plan(move, [
            "configure_device requires non-empty param_overrides OR a "
            "preset+device_slug combination that resolves to params"
        ])

    # Wire-format keys: the remote_command backend bypasses the MCP
    # tool's normalization layer, so emit name_or_index directly
    # (see mcp_server/semantic_moves/device_mutation_compilers.py
    # original comment for full context).
    parameters = [
        {"name_or_index": str(name), "value": value}
        for name, value in merged.items()
    ]

    step = CompiledStep(
        tool="batch_set_parameters",
        params={
            "track_index": track_index,
            "device_index": device_index,
            "parameters": parameters,
        },
        description=(
            f"Configure device at track {track_index}, device_index {device_index}"
            + (f" from preset {device_slug}/{preset_name}" if preset_name else "")
            + f" — set {len(parameters)} param(s)"
        ),
        verify_after=True,
        backend="remote_command",
    )

    return CompiledPlan(
        move_id=move.move_id,
        intent=move.intent,
        steps=[step],
        risk_level=move.risk_level,
        summary=step.description,
        requires_approval=(kernel.get("mode", "improve") != "explore"),
        warnings=[],
    )

"""Audit test: every module that imports SessionLedger or reads
memory_list / get_session_memory must annotate its use with a
store_purpose comment. This prevents the v1.20 store-confusion bug
(memory_list used for anti-repetition when it reads the persistent
technique library, not the action ledger) from recurring silently.

Invariants enforced:
  * Any file that imports SessionLedger or reads action_ledger has a
    comment near the read site with `store_purpose: <purpose>`.
  * Any file that calls memory_list() has a comment naming it as
    `store_purpose: technique_library` (not anti_repetition).
  * Any file that calls get_session_memory() has a comment naming it
    as `store_purpose: session_observations` or `escape_hatch_log`.
"""

from __future__ import annotations

import re
from pathlib import Path

import pytest


_MCP_SERVER = Path(__file__).resolve().parent.parent / "mcp_server"


_READER_PATTERNS: dict[str, str] = {
    # pattern name -> regex matched against each python file's text
    "SessionLedger_import": r"from\s+[\w.]+action_ledger\s+import\s+SessionLedger",
    "action_ledger_ctx_read": r'lifespan_context[^\n]+["\']action_ledger["\']',
    "memory_list_call": r"\bmemory_list\s*\(",
    "get_session_memory_call": r"\bget_session_memory\s*\(",
}


def _all_python_files() -> list[Path]:
    return [
        p for p in _MCP_SERVER.rglob("*.py")
        if "__pycache__" not in p.parts
    ]


def _has_purpose_comment_near(text: str, match_pos: int, window: int = 400) -> bool:
    """Check if a `# store_purpose: ...` comment appears within `window`
    chars before or after the matched line.

    The comment must be a single-line Python comment with the exact
    prefix `store_purpose:` (case-insensitive) — this is the contract.
    """
    start = max(0, match_pos - window)
    end = min(len(text), match_pos + window)
    snippet = text[start:end]
    return bool(re.search(
        r"#\s*store_purpose\s*:\s*\w+",
        snippet,
        re.IGNORECASE,
    ))


class TestLedgerReaderAnnotations:
    @pytest.mark.parametrize("pattern_name,pattern", list(_READER_PATTERNS.items()))
    def test_every_hit_has_store_purpose_comment(self, pattern_name, pattern):
        failures: list[str] = []
        regex = re.compile(pattern)
        for path in _all_python_files():
            # Exempt the definition files themselves — they're the
            # origin of the surface, not consumers.
            if path.name in ("action_ledger.py", "action_tools.py"):
                continue
            # Tests also exempted — they're the invariant enforcer.
            if "/tests/" in str(path):
                continue
            text = path.read_text(encoding="utf-8")
            for match in regex.finditer(text):
                if not _has_purpose_comment_near(text, match.start()):
                    # Determine the line number of the match
                    line_no = text[:match.start()].count("\n") + 1
                    failures.append(
                        f"{path.relative_to(_MCP_SERVER.parent)}:{line_no} "
                        f"{pattern_name} hit without nearby `# store_purpose: <purpose>` comment"
                    )
        assert not failures, (
            "Store-purpose audit failures:\n  " + "\n  ".join(failures)
            + "\n\nAdd a `# store_purpose: <purpose>` comment near each hit. "
            "Allowed purposes: anti_repetition | technique_library | "
            "session_observations | escape_hatch_log | audit_readonly | writer."
        )


class TestAntiRepetitionUsesLedgerNotMemoryList:
    """Anti-repetition paths must NOT call memory_list. The store-confusion
    bug in v1.20 came from docs pointing at memory_list for recency when
    it actually reads the persistent technique library."""

    def test_no_anti_repetition_annotated_file_calls_memory_list(self):
        """If a file's store_purpose says anti_repetition, it must not
        ALSO call memory_list() for recency. memory_list is for the
        technique library only."""
        failures: list[str] = []
        anti_rep_regex = re.compile(
            r"#\s*store_purpose\s*:\s*anti_repetition",
            re.IGNORECASE,
        )
        memory_list_regex = re.compile(r"\bmemory_list\s*\(")
        for path in _all_python_files():
            if "/tests/" in str(path):
                continue
            text = path.read_text(encoding="utf-8")
            if not anti_rep_regex.search(text):
                continue
            if memory_list_regex.search(text):
                failures.append(
                    f"{path.relative_to(_MCP_SERVER.parent)}: "
                    f"annotated anti_repetition but also calls memory_list() "
                    f"— memory_list reads the persistent technique library, "
                    f"not the action ledger."
                )
        assert not failures, (
            "Anti-repetition / memory_list conflict:\n  " + "\n  ".join(failures)
        )
